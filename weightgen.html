def from_binomial(
    p: float,
    reservoir_dimensionality: int, 
    n: int = 1, 
    input_dimensionality: int = 0,
    gen: Optional[np.random.Generator]=None
) -> np.typing.NDArray[np.integer]: 
    """
    Generate a weight matrix from a binomial distribution 

    Params: 

    p: float
        Probability of success of each trial. Between 0 to 1 

    reservoir_dimensionality: int
        Number of nodes in the reservoir

    n: int 
        Number of trials. Defaults to 1 (Bernoulli Distribution)
    
    input_dimensionality: int 
        Dimensionality of input vector. Defaults to 0 (Not W_in)

    gen: np.random.Generator 
        NumPy Random Number Generator object. Defaults to None
    """
    
    if not gen: 
        gen = np.random.default_rng(seed=0)
    
    if not input_dimensionality:
        return gen.binomial(
            n=n, 
            p=p,
            size=(reservoir_dimensionality, reservoir_dimensionality)
        )
    
    return gen.binomial(
        n=n,
        p=p,
        size=(reservoir_dimensionality, input_dimensionality)
    )

) -> np.typing.NDArray[np.floating]: 
    """
    Generate a weight matrix from a uniform distribution

        Lower bound of random numbers

def make_sparse(
    weight_matrix: np.typing.NDArray[np.floating], 
    sparsity: float
) -> np.typing.NDArray[np.floating]:
    """
    Make a weight matrix sparse 

    Params: 

    weight_matrix: np.typing.NDArray[np.floating]
        Weight matrix to be made sparse
    
    sparsity: float 
        Extent of how sparse to make the weight matrix. Ranges from 0 to 1.
    """

    return weight_matrix * (np.random.rand(*weight_matrix.shape) < sparsity)

def set_spectral(
    weight_matrix: np.typing.NDArray[np.floating], 
    spectral_radius: float
) -> np.typing.NDArray[np.floating]:
    
    """
    Set the spectral radius of the weight matrix 

    Params: 

    weight_matrix: np.typing.NDArray[np.floating]
        Weight matrix whose spectral radius is to be set

    spectral_radius: float
        p(W) The largest absolute eigenvalue of the weight matrix. 
        Values less than 1.0 are recommended for tasks that require significant memory fading. 
        Values between 1-1.5 are recommended for takes that are memory dependent. 
    """

    return spectral_radius/np.linalg.norm(weight_matrix, ord=2) * weight_matrix
